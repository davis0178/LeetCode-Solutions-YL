```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        node0 = dummy = ListNode(next=head)
        node1 = head

        while node1 and node1.next: # 确保有大于等于两个待交换的节点
            # 先拿到 node2 node3
            node2 = node1.next
            node3 = node2.next
			
			# 按顺序连接
            node0.next = node2
            node2.next = node1
            node1.next = node3
			
			# 初始化下一个过程
            node0 = node1
            node1 = node3
        return dummy.next
```
