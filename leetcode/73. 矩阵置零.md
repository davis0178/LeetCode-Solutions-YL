```python
class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """
        m, n = len(matrix), len(matrix[0])
        flag_row0 = 0 in matrix[0]
        flag_col0 = any(matrix[i][0] == 0 for i in range(m))

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][j] == 0:
                    matrix[0][j] = 0
                    matrix[i][0] = 0
        
        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][0] == 0 or matrix[0][j] == 0:
                    matrix[i][j] = 0
        
        if flag_row0:
            for j in range(n):
                matrix[0][j] = 0
        
        if flag_col0:
            for i in range(m):
                matrix[i][0] = 0
```

先储存第一行第一列是否有 0
第一次遍历：将剩余行列的 0 存在第一行第一列
第二次遍历：根据第一行第一列是否有 0 将剩余矩阵置零
最后，根据储存的第一行第一列是否有 0 判断是否将第一行第一列置为 0

这里第一次第二次遍历都只需遍历**除第一行第一列剩下的行列**，不需要遍历第一行第一列
