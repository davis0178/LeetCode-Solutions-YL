```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        fast = slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast is slow:
                while head is not slow:
                    head = head.next
                    slow = slow.next
                return head
        
        return None
```

还是快慢指针。但是如果只是快慢相遇的话，不一定在入环口。

**Floyd 判圈算法**：设 head 到入环口需要走 a 步，环长为 c。快慢指针相遇后，数学等式表明：**head 和 slow再同时走 a 步**，就恰好到入环口了。
