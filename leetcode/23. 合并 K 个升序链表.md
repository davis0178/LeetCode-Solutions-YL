```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = cur = ListNode()
        while l1 and l2:
            if l1.val <= l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        cur.next = l1 or l2
        return dummy.next

    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        m = len(lists)
        if m == 0:
            return None
        if m == 1:
            return lists[0]
        
        left = self.mergeKLists(lists[:m//2])
        right = self.mergeKLists(lists[m//2:])
        return self.mergeTwoLists(left, right)
```

这里也是和[[148. 排序链表]]一样的思路，先按照中点分为两端，然后递归出口决定了当参数为单个链表或None时进入合并，将合并 K 个升序链表的大问题不断拆成合并 K / 2 个升序链表的小问题，最终合成最终结果
